30 inline Periodic::Periodic(
31  const xt::xtensor<double, 2>& coor,
32  const xt::xtensor<size_t, 2>& dofs,
33  const xt::xtensor<size_t, 2>& control,
34  const xt::xtensor<size_t, 2>& nodal_tyings,

41  GOOSEFEM_ASSERT(xt::has_shape(m_tyings, {m_nties, size_t(2)}));
42  GOOSEFEM_ASSERT(xt::has_shape(control, {m_ndim, m_ndim}));
43  GOOSEFEM_ASSERT(xt::has_shape(dofs, m_coor.shape()));
44  GOOSEFEM_ASSERT(xt::amax(control)() <= xt::amax(dofs)());
45  GOOSEFEM_ASSERT(xt::amin(control)() >= xt::amin(dofs)());
46  GOOSEFEM_ASSERT(xt::amax(iip)() <= xt::amax(dofs)());
47  GOOSEFEM_ASSERT(xt::amin(iip)() >= xt::amin(dofs)());
48  GOOSEFEM_ASSERT(xt::amax(nodal_tyings)() < m_coor.shape(0));
49 
50  xt::xtensor<size_t, 1> dependent = xt::view(m_tyings, xt::all(), 1);
51  xt::xtensor<size_t, 2> dependent_dofs = xt::view(dofs, xt::keep(dependent), xt::all());
52  xt::xtensor<size_t, 1> iid = xt::flatten(dependent_dofs);
53  xt::xtensor<size_t, 1> iii = xt::setdiff1d(dofs, iid);
54  xt::xtensor<size_t, 1> iiu = xt::setdiff1d(iii, iip);
55 
56  m_nnu = iiu.size();
57  m_nnp = iip.size();
58  m_nni = iii.size();
59  m_nnd = iid.size();
60 
61  GooseFEM::Mesh::Reorder reorder({iiu, iip, iid});
62 
63  m_dofs = reorder.apply(dofs);
64  m_control = reorder.apply(control);
65 }
