81  GOOSEFEM_ASSERT(Element::isDiagonal(elemmat));
82 
83  m_Auu.fill(0.0);
84  m_App.fill(0.0);
85 
86  for (size_t e = 0; e < m_nelem; ++e) {

89

91

93  m_Auu(d) += elemmat(e, m * m_ndim + i, m * m_ndim + i);
94  }
95  else {
96  m_App(d - m_nnu) += elemmat(e, m * m_ndim + i, m * m_ndim + i);
97  }
98  }
99  }
