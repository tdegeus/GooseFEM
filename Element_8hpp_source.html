335 template <class D>
336 inline void QuadratureBaseCartesian<D>::compute_dN_impl()
337 {
338  auto nelem = derived_cast().m_nelem;
339  auto nip = derived_cast().m_nip;
340  auto& vol = derived_cast().m_vol;
341  auto& w = derived_cast().m_w;
342  auto& dNxi = derived_cast().m_dNxi;
343  auto& dNx = derived_cast().m_dNx;
344  auto& x = derived_cast().m_x;
345 
346  dNx.fill(0.0);
347 
348  #pragma omp parallel
349  {
350  auto J = xt::xtensor<double, 2>::from_shape({D::s_ndim, D::s_ndim});
351  auto Jinv = xt::xtensor<double, 2>::from_shape({D::s_ndim, D::s_ndim});
352 
353  #pragma omp for
354  for (size_t e = 0; e < nelem; ++e) {
355 
356  auto xe = xt::adapt(&x(e, 0, 0), xt::xshape<D::s_nne, D::s_ndim>());
357 
358  for (size_t q = 0; q < nip; ++q) {
359 
360  auto dNxiq = xt::adapt(&dNxi(q, 0, 0), xt::xshape<D::s_nne, D::s_ndim>());
361  auto dNxq = xt::adapt(&dNx(e, q, 0, 0), xt::xshape<D::s_nne, D::s_ndim>());
362 
363  J.fill(0.0);
364 
365  for (size_t m = 0; m < D::s_nne; ++m) {
366  for (size_t i = 0; i < D::s_ndim; ++i) {
367  for (size_t j = 0; j < D::s_ndim; ++j) {
368  J(i, j) += dNxiq(m, i) * xe(m, j);
369  }
370  }
371  }
372 
373  double Jdet = detail::tensor<D::s_ndim>::inv(J, Jinv);
374 
375  for (size_t m = 0; m < D::s_nne; ++m) {
376  for (size_t i = 0; i < D::s_ndim; ++i) {
377  for (size_t j = 0; j < D::s_ndim; ++j) {
378  dNxq(m, i) += Jinv(i, j) * dNxiq(m, i);
379  }
380  }
381  }
382 
383  vol(e, q) = w(q) * Jdet;
384  }
385  }
386  }

633  GOOSEFEM_ASSERT(xt::has_shape(qvector, this->shape_qvector(n)));

639  auto& vol = derived_cast().m_vol;
640 
641  elemvec.fill(0.0);
642 
643  #pragma omp parallel for
644  for (size_t e = 0; e < nelem; ++e) {
645

655  for (size_t i = 0; i < n; ++i) {
