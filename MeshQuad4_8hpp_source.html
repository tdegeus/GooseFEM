609  // Allocate output
610 
611  size_t N = 0;
612 
613  for (size_t i = yl; i <= yu; ++i) {
614  // no refinement
615  size_t n = (xu - xl) * hx / m_nhx(i);
616  // refinement
617  if (m_refine(i) != -1) {
618  n *= 4;
619  }
620  N += n;
621  }
622 
623  xt::xtensor<size_t, 1> ret = xt::empty<size_t>({N});
624 
625  // Write output
626 
627  N = 0;
628 
629  for (size_t i = yl; i <= yu; ++i) {
630  // no refinement
631  size_t n = (xu - xl) * hx / m_nhx(i);
632  size_t h = hx;
633  // refinement
634  if (m_refine(i) != -1) {
635  n *= 4;
636  h *= 4;
637  }
638  xt::xtensor<size_t, 1> e = m_startElem(i) + xl * h / m_nhx(i) + xt::arange<size_t>(n);
639  xt::view(ret, xt::range(N, N + n)) = e;
640  N += n;
641  }
642

652

684  xu = mid + size + 1;

887 {

891

893  for (size_t iy = 0; iy < nely; ++iy) {
894 
895  // no refinement

898 
899  // refinement

905  // element numbers of the layer, and roll them

907  xt::view(ret, xt::range(m_startElem(iy), m_startElem(iy) + nel)) = xt::roll(e, shift);

913 template <class C, class E>
914 inline void FineLayer::map(const C& coor, const E& conn)
915 {
916  GOOSEFEM_ASSERT(coor.dimension() == 2);
917  GOOSEFEM_ASSERT(conn.dimension() == 2);
918  GOOSEFEM_ASSERT(coor.shape(1) == 2);
919  GOOSEFEM_ASSERT(conn.shape(1) == 4);
920  GOOSEFEM_ASSERT(conn.shape(0) > 0);
921  GOOSEFEM_ASSERT(coor.shape(0) >= 4);
922  GOOSEFEM_ASSERT(xt::amax(conn)() < coor.shape(0));
923 
924  if (conn.shape(0) == 1) {
925  this->init(1, 1, coor(conn(0, 1), 0) - coor(conn(0, 0), 0));
926  GOOSEFEM_CHECK(xt::all(xt::equal(this->conn(), conn)));
927  GOOSEFEM_CHECK(xt::allclose(this->coor(), coor));
928  return;
929  }
930 
931  // Identify the middle layer
932 
933  size_t emid = (conn.shape(0) - conn.shape(0) % 2) / 2;
934  size_t eleft = emid;
935  size_t eright = emid;
936 
937  while (conn(eleft, 0) == conn(eleft - 1, 1) && eleft > 0) {
938  eleft--;
939  }
940 
941  while (conn(eright, 1) == conn(eright + 1, 0) && eright < conn.shape(0) - 1) {
942  eright++;
943  }
944 
945  GOOSEFEM_CHECK(xt::allclose(coor(conn(eleft, 0), 0), 0.0));
946 
947  // Get element sizes along the middle layer
948 
949  auto n0 = xt::view(conn, xt::range(eleft, eright + 1), 0);
950  auto n1 = xt::view(conn, xt::range(eleft, eright + 1), 1);
951  auto n2 = xt::view(conn, xt::range(eleft, eright + 1), 2);
952  auto dx = xt::view(coor, xt::keep(n1), 0) - xt::view(coor, xt::keep(n0), 0);
953  auto dy = xt::view(coor, xt::keep(n2), 1) - xt::view(coor, xt::keep(n1), 1);
954  auto hx = xt::amin(dx)();
955  auto hy = xt::amin(dy)();
956 
957  GOOSEFEM_CHECK(xt::allclose(hx, hy));
958  GOOSEFEM_CHECK(xt::allclose(dx, hx));
959  GOOSEFEM_CHECK(xt::allclose(dy, hy));
960 
961  // Extract shape and initialise
962 
963  size_t nelx = eright - eleft + 1;
964  size_t nely = static_cast<size_t>((coor(coor.shape(0) - 1, 1) - coor(0, 1)) / hx);
965  this->init(nelx, nely, hx);
966  GOOSEFEM_CHECK(xt::all(xt::equal(this->conn(), conn)));
967  GOOSEFEM_CHECK(xt::allclose(this->coor(), coor));
968  GOOSEFEM_CHECK(xt::all(xt::equal(this->elementsMiddleLayer(), eleft + xt::arange<size_t>(nelx))));
969 }

1081  : m_finelayer(mesh)
1082 {

1084  // Regular-mesh
1085  // ------------
1086 
1087  m_regular = GooseFEM::Mesh::Quad4::Regular(
1088  xt::amax(m_finelayer.m_layer_nelx)(), xt::sum(m_finelayer.m_nhy)(), m_finelayer.m_h);
1089 
1090  // -------
1091  // mapping
1092  // -------
1093 
1094  // allocate mapping
1095  m_elem_regular.resize(m_finelayer.m_nelem);
1096  m_frac_regular.resize(m_finelayer.m_nelem);
1097 
1098  // alias
1099  xt::xtensor<size_t, 1> nhx = m_finelayer.m_nhx;
1100  xt::xtensor<size_t, 1> nhy = m_finelayer.m_nhy;
1101  xt::xtensor<size_t, 1> nelx = m_finelayer.m_layer_nelx;
1102  xt::xtensor<size_t, 1> start = m_finelayer.m_startElem;
1103 
1104  // 'matrix' of element numbers of the Regular-mesh
1105  xt::xtensor<size_t, 2> elementgrid = m_regular.elementgrid();
1106 
1107  // cumulative number of element-rows of the Regular-mesh per layer of the FineLayer-mesh
1108  xt::xtensor<size_t, 1> cum_nhy =
1109  xt::concatenate(xt::xtuple(xt::zeros<size_t>({1}), xt::cumsum(nhy)));
1110 
1111  // number of element layers in y-direction of the FineLayer-mesh
1112  size_t nely = nhy.size();
1113 
1114  // loop over layers of the FineLayer-mesh
1115  for (size_t iy = 0; iy < nely; ++iy) {
1116  // element numbers of the Regular-mesh along this layer of the FineLayer-mesh
1117  auto el_new = xt::view(elementgrid, xt::range(cum_nhy(iy), cum_nhy(iy + 1)), xt::all());
1118 
1119  // no coarsening/refinement
1120  // ------------------------
1121 
1122  if (m_finelayer.m_refine(iy) == -1) {
1123  // element numbers of the FineLayer-mesh along this layer
1124  xt::xtensor<size_t, 1> el_old = start(iy) + xt::arange<size_t>(nelx(iy));
1125 
1126  // loop along this layer of the FineLayer-mesh
1127  for (size_t ix = 0; ix < nelx(iy); ++ix) {
1128  // get the element numbers of the Regular-mesh for this element of the
1129  // FineLayer-mesh
1130  auto block =
1131  xt::view(el_new, xt::all(), xt::range(ix * nhx(iy), (ix + 1) * nhx(iy)));
1132 
1133  // write to mapping
1134  for (auto& i : block) {
1135  m_elem_regular[el_old(ix)].push_back(i);
1136  m_frac_regular[el_old(ix)].push_back(1.0);
1137  }
1138  }
1139  }
1140 
1141  // refinement along the x-direction (below the middle layer)
1142  // ---------------------------------------------------------
1143 
1144  else if (m_finelayer.m_refine(iy) == 0 && iy <= (nely - 1) / 2) {
1145  // element numbers of the FineLayer-mesh along this layer
1146  // rows: coarse block, columns element numbers per block
1147  xt::xtensor<size_t, 2> el_old =
1148  start(iy) + xt::arange<size_t>(nelx(iy) * 4ul).reshape({-1, 4});
1149 
1150  // loop along this layer of the FineLayer-mesh
1151  for (size_t ix = 0; ix < nelx(iy); ++ix) {
1152  // get the element numbers of the Regular-mesh for this block of the FineLayer-mesh
1153  auto block =
1154  xt::view(el_new, xt::all(), xt::range(ix * nhx(iy), (ix + 1) * nhx(iy)));
1155 
1156  // bottom: wide-to-narrow
1157  {
1158  for (size_t j = 0; j < nhy(iy) / 2; ++j) {
1159  auto e = xt::view(block, j, xt::range(j, nhx(iy) - j));
1160 
1161  m_elem_regular[el_old(ix, 0)].push_back(e(0));
