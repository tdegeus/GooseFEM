67 inline void QuadratureAxisymmetric::compute_dN_impl()
68 {
69  // most components remain zero, and are not written
70  m_B.fill(0.0);
71 
72  #pragma omp parallel
73  {
74  xt::xtensor<double, 2> J = xt::empty<double>({2, 2});
75  xt::xtensor<double, 2> Jinv = xt::empty<double>({2, 2});
76 
77  #pragma omp for
78  for (size_t e = 0; e < m_nelem; ++e) {
79 
80  auto x = xt::adapt(&m_x(e, 0, 0), xt::xshape<s_nne, s_ndim>());
81 
82  for (size_t q = 0; q < m_nip; ++q) {
83 
84  auto dNxi = xt::adapt(&m_dNxi(q, 0, 0), xt::xshape<s_nne, s_ndim>());
85  auto B = xt::adapt(&m_B(e, q, 0, 0, 0, 0), xt::xshape<s_nne, s_tdim, s_tdim, s_tdim>());
86  auto N = xt::adapt(&m_N(q, 0), xt::xshape<s_nne>());
87 
88  // J(i,j) += dNxi(m,i) * x(m,j);
89  J(0, 0) = dNxi(0, 0) * x(0, 0) + dNxi(1, 0) * x(1, 0) + dNxi(2, 0) * x(2, 0) +
90  dNxi(3, 0) * x(3, 0);
91  J(0, 1) = dNxi(0, 0) * x(0, 1) + dNxi(1, 0) * x(1, 1) + dNxi(2, 0) * x(2, 1) +
92  dNxi(3, 0) * x(3, 1);
93  J(1, 0) = dNxi(0, 1) * x(0, 0) + dNxi(1, 1) * x(1, 0) + dNxi(2, 1) * x(2, 0) +
94  dNxi(3, 1) * x(3, 0);
95  J(1, 1) = dNxi(0, 1) * x(0, 1) + dNxi(1, 1) * x(1, 1) + dNxi(2, 1) * x(2, 1) +
96  dNxi(3, 1) * x(3, 1);
97 
98  double Jdet = detail::tensor<2>::inv(J, Jinv);
99 
100  // radius for computation of volume
101  double rq = N(0) * x(0, 1) + N(1) * x(1, 1) + N(2) * x(2, 1) + N(3) * x(3, 1);
102 
103  // dNx(m,i) += Jinv(i,j) * dNxi(m,j)
104  for (size_t m = 0; m < s_nne; ++m) {
105  // B(m, r, r, r) = dNdx(m,1)
106  B(m, 0, 0, 0) = Jinv(1, 0) * dNxi(m, 0) + Jinv(1, 1) * dNxi(m, 1);
107  // B(m, r, z, z) = dNdx(m,1)
108  B(m, 0, 2, 2) = Jinv(1, 0) * dNxi(m, 0) + Jinv(1, 1) * dNxi(m, 1);
109  // B(m, t, t, r)
110  B(m, 1, 1, 0) = 1.0 / rq * N(m);
111  // B(m, z, r, r) = dNdx(m,0)
112  B(m, 2, 0, 0) = Jinv(0, 0) * dNxi(m, 0) + Jinv(0, 1) * dNxi(m, 1);
113  // B(m, z, z, z) = dNdx(m,0)
114  B(m, 2, 2, 2) = Jinv(0, 0) * dNxi(m, 0) + Jinv(0, 1) * dNxi(m, 1);
115  }
116 
117  m_vol(e, q) = m_w(q) * Jdet * 2.0 * M_PI * rq;
118  }
119  }
120  }
